LeetCode简单题：1.两数之和（Python,C++,Java）

 Posted on August 8, 2020  |  1 minutes  |  167 words  |  Kun Qian

一.解法

https://leetcode-cn.com/problems/two-sum/

要点：hashmap

三种语言都用了hash map

Python 中用hashmap={}

C++中用unordered_map<int, int> hashmap;

Java中用Map<Integer, Integer> hashmap = new HashMap<>();

二.Python实现

def twoSum(nums, target):
    hashmap={}
    for ind,num in enumerate(nums):
        hashmap[num] = ind
    for i,num in enumerate(nums):
        j = hashmap.get(target - num)
        if j is not None and i!=j:
            return [i,j]

三.C++实现

class Solution {
public:
	vector<int> twoSum(vector<int>& nums, int target) {
		unordered_map<int, int> hashmap;
		vector<int> result;
		for (int i = 0; i < nums.size(); i++) {
			hashmap[nums[i]] = i;
		}
		for (int i = 0; i < nums.size(); i++) {
			if (hashmap[target - nums[i]] && (hashmap[target - nums[i]] != i)) {
				result.push_back(i);
				result.push_back(hashmap[target - nums[i]]);
				return result;
			}
		}
		return result;
	}
};

四.java实现

class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> hashmap = new HashMap<>();
         for(int i = 0; i< nums.length; i++) {
            hashmap.put(nums[i], i);
        }
        for(int i = 0; i< nums.length; i++) {
            if(hashmap.containsKey(target - nums[i])) {
                if(hashmap.get(target-nums[i])!=i){
                    if(hashmap.get(target-nums[i])>i){
                        return new int[] {i,hashmap.get(target-nums[i])};
                    }
                return new int[] {hashmap.get(target-nums[i]),i};}
            }
        }
       return new int[] {-1,-1};
    }}
