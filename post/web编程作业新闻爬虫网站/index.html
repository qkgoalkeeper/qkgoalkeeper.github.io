search.html

总体思路是通过几个按钮将关键词传到后台访问MySQL数据库，再将数据返回到页面。

2.后端设计

mysql.js采用课上提供的版本，用于连接MySQL数据库，代码如下：

var mysql = require("mysql");
var pool = mysql.createPool({
    host: '127.0.0.1',
    user: 'root',
    password: 'root',
    database: 'crawl'
});
var query = function(sql, sqlparam, callback) {
    pool.getConnection(function(err, conn) {
        if (err) {
            callback(err, null, null);
        } else {
            conn.query(sql, sqlparam, function(qerr, vals, fields) {
                //conn.release(); //释放连接 
                pool.releaseConnection(conn); 
                callback(qerr, vals, fields); //事件驱动回调 
            });
        }
    });
};
var query_noparam = function(sql, callback) {
    pool.getConnection(function(err, conn) {
        if (err) {
            callback(err, null, null);
        } else {
            conn.query(sql, function(qerr, vals, fields) {
                //conn.release(); //释放连接 
                pool.releaseConnection(conn); 
                callback(qerr, vals, fields); //事件驱动回调 
            });
        }
    });
};
exports.query = query;
exports.query_noparam = query_noparam;

var express = require('express');
var router = express.Router();
var mysql = require('../mysql.js')

/* GET home page. */
router.get('/', function(req, res, next) {
  res.render('index', { title: 'Express' });
});

router.get('/process_get', function(request, response){
  var use_or = false;//用来判断是否有多重查询
  var fetchSql = "select url,source_name,title,author,publish_date from fetches where";
  if (request.query.title_c=="true"){
    if(use_or == false){
      use_or = true;
    }
    fetchSql+=" title like '%"+request.query.search_w + "%'";
  }
  if (request.query.keywords_c=="true"){
    if(use_or == false){
      fetchSql+=" keywords like '%"+request.query.search_w + "%'";
      use_or = true;
    }
    else{fetchSql+=" or keywords like '%"+request.query.search_w + "%'";}
  }
  if (request.query.content_c=="true"){
    if(use_or == false){
      fetchSql+=" content like '%"+request.query.search_w + "%'";
      use_or = true;
    }
    else{fetchSql+=" or content like '%"+request.query.search_w + "%'";}
  }

  if(request.query.order_c=="true")fetchSql+=" order by publish_date DESC;";
  else fetchSql+=";";


  mysql.query(fetchSql, function(err, result, fields){
    response.writeHead(200, {
      "Content-Type": "application/json"
    });
    response.write(JSON.stringify(result));
    response.end();
  });

});




router.get('/process_time', function(request, response){
  
  var fetchSql = "select publish_date,count(*) from (select * from fetches where title like '%"+request.query.search_w+"%' or keywords like '%"+request.query.search_w+"%' or content like '%"+request.query.search_w+"%') as fetches2 group by publish_date ORDER BY publish_date DESC;"; 
  
  mysql.query(fetchSql, function(err, result, fields){
    response.writeHead(200, {
      "Content-Type": "application/json"
    });
    response.write(JSON.stringify(result));
    response.end();
  });

});

四.问题与总结

1.问题发现

①在做echart柱状图展示时一开始我采用循环访问数据库（见后端设计process_graph被注释的代码），发现MySQL的连接池有设置上限，短时反复查询数据库容易导致连接崩溃，后来修改sql查询语句设计了新的process_graph大大缓解了这个问题。

②目前的爬虫是我人工每天执行代码爬取的，如果部署在服务器上的话可以运用相关库每天定时爬取，更省时省力。

2.项目总结

整个项目总体上的框架与课上代码的思路类似，只不过自定义添加了一些功能。在爬虫部分中，项目完成了三个目标网站的分析和爬取，爬取了近3000条数据并存储在mysql数据库中。在搜索网站部分中，项目完成对数据库中爬取数据内容或标题或关键词的搜索功能，搜索结果以表格形式展示在前端页面中，也完成对搜索内容的时间热度分析（表格展示），额外完成了用echart模块实现时间热度分析柱状图可视化的功能，额外实现了搜索结果以时间排序展示的功能。最后，感谢老师和助教这几个月的指导，使我更加熟悉了HTML、CSS、JavaScript以及相关框架的使用。
